## Usage

```js
const remark = require('remark');
const crossref = require('remark-crossref');
const latex = require('remark-latex');

remark()
.use(crossref)
.use(latex)
.process(markdown, (err, vfile) => {
  if (err) {
    console.error(err);
  }
  console.log(vfile);
});
```
